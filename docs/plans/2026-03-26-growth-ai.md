      COALESCE(SUM(CASE WHEN d.stage_name = 'Negócio Ganho' THEN COALESCE(d.valor_recorrente::numeric, 0) ELSE 0 END), 0)::numeric(12,2) as faturamento_recorrente,

      COUNT(DISTINCT d.fnl_ngc) as funis_distintos

async function getBudgets(db: any, input: any): Promise<string> {
  const { startDate, endDate, funil = 'todos' } = input;

  // Generate months in range
  const meses: string[] = [];
  const start = new Date(startDate + '-01');
  const end = new Date(endDate + '-01');
  const cursor = new Date(start);
  while (cursor <= end) {
    const m = `${cursor.getFullYear()}-${String(cursor.getMonth() + 1).padStart(2, '0')}`;
    meses.push(m);
    cursor.setMonth(cursor.getMonth() + 1);
  }

  if (meses.length === 0) return JSON.stringify({ error: "Nenhum mês no range" });

  const result = await db.execute(sql`
    SELECT mes, segmento, metricas
    FROM meta_ads.growth_budgets
    WHERE mes IN (${sql.join(meses.map(m => sql`${m}`), sql`, `)})
      AND funil = ${funil}
    ORDER BY mes, segmento
  `);

  return JSON.stringify(result.rows);
}

async function getCampaignRanking(db: any, input: any): Promise<string> {
  const { startDate, endDate, metric, order = 'top', limit = 10 } = input;

  const metricExpr: Record<string, string> = {
    spend: 'SUM(mid.spend)',
    cpc: 'CASE WHEN SUM(mid.clicks) > 0 THEN SUM(mid.spend) / SUM(mid.clicks) ELSE 0 END',
    ctr: 'CASE WHEN SUM(mid.impressions) > 0 THEN SUM(mid.inline_link_clicks)::float / SUM(mid.impressions) * 100 ELSE 0 END',
    impressions: 'SUM(mid.impressions)',
    clicks: 'SUM(mid.clicks)',
    cpm: 'CASE WHEN SUM(mid.impressions) > 0 THEN SUM(mid.spend) / SUM(mid.impressions) * 1000 ELSE 0 END',
  };

  const expr = metricExpr[metric] || 'SUM(mid.spend)';
  const orderDir = order === 'bottom' ? 'ASC' : 'DESC';
  const safeLimit = Math.min(Math.max(1, limit), 50);

  const result = await db.execute(sql.raw(`
    SELECT c.campaign_name,
      (${expr})::numeric(12,2) as valor,
      SUM(mid.spend)::numeric(12,2) as investimento,
      SUM(mid.impressions) as impressoes,
      SUM(mid.clicks) as cliques,
      SUM(mid.inline_link_clicks) as cliques_saida
    FROM meta_ads.meta_insights_daily mid
    JOIN meta_ads.meta_campaigns c ON c.campaign_id = mid.campaign_id
    WHERE mid.date_start >= '${startDate}'::date AND mid.date_start <= '${endDate}'::date
      AND mid.account_id = '${TURBO_ACCOUNT_ID}'
      AND mid.spend > 0
    GROUP BY c.campaign_name
    HAVING ${expr} > 0
    ORDER BY valor ${orderDir}
    LIMIT ${safeLimit}
  `));

  return JSON.stringify(result.rows);
}

async function getHealthCheck(db: any, input: any): Promise<string> {
  const { startDate, endDate } = input;

  // 1. Resumo geral de ads
  const adsTotal = await getAdsMetrics(db, { startDate, endDate, groupBy: 'total' });

  // 2. Resumo de deals
  const dealsTotal = await getDealsMetrics(db, { startDate, endDate });

  // 3. Performance por campanha (top spenders)
  const topCampaigns = await getCampaignRanking(db, { startDate, endDate, metric: 'spend', order: 'top', limit: 15 });

  // 4. Período anterior (mesmo range antes)
  const startD = new Date(startDate);
  const endD = new Date(endDate);
  const diffDays = Math.ceil((endD.getTime() - startD.getTime()) / (1000 * 60 * 60 * 24));
  const prevEnd = new Date(startD);
  prevEnd.setDate(prevEnd.getDate() - 1);
  const prevStart = new Date(prevEnd);
  prevStart.setDate(prevStart.getDate() - diffDays);
  const prevAds = await getAdsMetrics(db, {
    startDate: prevStart.toISOString().split('T')[0],
    endDate: prevEnd.toISOString().split('T')[0],
    groupBy: 'total'
  });
  const prevDeals = await getDealsMetrics(db, {
    startDate: prevStart.toISOString().split('T')[0],
    endDate: prevEnd.toISOString().split('T')[0],
  });

  return JSON.stringify({
    periodo: { startDate, endDate },
    periodo_anterior: { startDate: prevStart.toISOString().split('T')[0], endDate: prevEnd.toISOString().split('T')[0] },
    ads_atual: JSON.parse(adsTotal),
    deals_atual: JSON.parse(dealsTotal),
    ads_anterior: JSON.parse(prevAds),
    deals_anterior: JSON.parse(prevDeals),
    top_campanhas: JSON.parse(topCampaigns),
  });
}

      // Save user message
      await db.execute(sql`
        INSERT INTO cortex_core.growth_ai_mensagens (conversa_id, role, conteudo)
        VALUES (${conversaId}, 'user', ${message})
      `);

      // Load history
      const histResult = await db.execute(sql`
        SELECT role, conteudo FROM cortex_core.growth_ai_mensagens
        WHERE conversa_id = ${conversaId}
        ORDER BY criado_em ASC
      `);
      const messages = histResult.rows.map((m: any) => ({
        role: m.role as "user" | "assistant",
        content: m.conteudo,
      }));

      // Tool use loop
      let currentMessages: any[] = messages;
      let finalResponse = "";
      let allToolCalls: any[] = [];

      for (let attempt = 0; attempt < 5; attempt++) {
        const response = await anthropic.messages.create({
          model: MODEL,
          max_tokens: 8192,
          system: SYSTEM_PROMPT,
          tools: GROWTH_AI_TOOLS as any,
          messages: currentMessages,
        });

        // Check if model wants to use tools
        const toolUseBlocks = response.content.filter((b: any) => b.type === "tool_use");
        const textBlocks = response.content.filter((b: any) => b.type === "text");

        if (toolUseBlocks.length === 0) {
          // No tools — final text response
          finalResponse = textBlocks.map((b: any) => b.text).join("\n");
          break;
        }

        // Execute tools
        const toolResults: any[] = [];
        for (const toolBlock of toolUseBlocks) {
          const result = await executeGrowthTool(db, toolBlock.name, toolBlock.input);
          allToolCalls.push({ tool: toolBlock.name, input: toolBlock.input, output_preview: result.substring(0, 200) });
          toolResults.push({
            type: "tool_result",
            tool_use_id: toolBlock.id,
            content: result,
          });
        }

        // Add assistant response + tool results to conversation
        currentMessages = [
          ...currentMessages,
          { role: "assistant", content: response.content },
          { role: "user", content: toolResults },
        ];
      }

      // Save assistant response
      await db.execute(sql`
        INSERT INTO cortex_core.growth_ai_mensagens (conversa_id, role, conteudo, tool_calls)
        VALUES (${conversaId}, 'assistant', ${finalResponse}, ${JSON.stringify(allToolCalls)}::jsonb)
      `);

      // Auto-title
      const conversa = conv.rows[0] as any;
      if (conversa.titulo === 'Nova conversa') {
        const shortTitle = message.length > 60 ? message.substring(0, 57) + "..." : message;
        await db.execute(sql`
          UPDATE cortex_core.growth_ai_conversas
          SET titulo = ${shortTitle}, atualizado_em = NOW()
          WHERE id = ${conversaId}
        `);
      } else {
        await db.execute(sql`
          UPDATE cortex_core.growth_ai_conversas SET atualizado_em = NOW() WHERE id = ${conversaId}
        `);
      }

      res.json({ role: "assistant", conteudo: finalResponse, conversaId, toolCalls: allToolCalls });
    } catch (error: any) {
      console.error("[growth-ai] Erro no chat:", error);
      res.status(500).json({ message: error.message || "Erro no chat" });
    }
  });
}
